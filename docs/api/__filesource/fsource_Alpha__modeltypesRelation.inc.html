  386.         global $config;
  387.         
  388.         if($this->relationType == 'MANY-TO-MANY') {
  389.             /*

  395.             if(empty($this->lookup)) {

  400.             if(empty($accessingClassName)) {

  411.

  413.                 foreach($lookupObjects as $lookupObject) {

  418.                 asort($values);

  420.             }
  421.             // load objects on the left from accessing on the right

  430.                     $obj->load($lookupObject->get('leftID'));
  431.                     array_push($values, $obj->get($this->relatedClassLeftDisplayField));
  432.                 }
  433.                 // sort array, then return as comma-seperated string
  434.                 asort($values);
  435.                 return implode(',', $values);
  436.             }
  437.         }else{

  458.         
  459.         if($this->relationType == 'ONE-TO-MANY') {
