  571.      * Deletes the current object from the database.
  572.      * 
  573.      * @since 1.0
  574.      * @throws FailedDeleteException
  575.      */
  576.     public function delete() {
  577.         self::$logger->debug('>>delete()');
  578.                 
  579.         if(method_exists($this, 'before_delete_callback'))
  580.                 $this->before_delete_callback();
  581.                 
  582.         global $config;
  583.         
  584.         // get the class attributes
  585.         $reflection = new ReflectionClass(get_class($this));
  586.         $properties = $reflection->getProperties();
  587.             
  588.         // check for any relations on this object, then remove them to prevent orphaned data
  589.         foreach($properties as $propObj) {
  590.             $propName = $propObj->name;
  591.                         
  592.             if(!$propObj->isPrivate() && isset($this->$propName) && $this->$propName instanceof Relation) {                
  593.                 $prop = $this->getPropObject($propName);
  594.                     
  595.                 // Handle MANY-TO-MANY rels
  596.                 if($prop->getRelationType() == 'MANY-TO-MANY') {
  597.                     self::$logger->debug('Deleting MANY-TO-MANY lookup objects...');
  598.                     
  599.                     try{
  600.                         // check to see if the rel is on this class
  601.                         $side = $prop->getSide(get_class($this));                                            
  602.                     }catch (IllegalArguementException $iae) {
  603.                         $side = $prop->getSide(ucfirst($this->getTableName()).'Object');
  604.                     }
  605.                                                                 
  606.                     self::$logger->debug('Side is ['.$side.']'.$this->getOID());
  607.                         
  608.                     $lookUp = $prop->getLookup();
  609.                     self::$logger->debug('Lookup object['.var_export($lookUp, true).']');
  610.                                         
  611.                     // delete all of the old RelationLookup objects for this rel
  612.                     if($side == 'left')
  613.                         $lookUp->deleteAllByAttribute('leftID', $this->getOID());
  614.                     else
  615.                         $lookUp->deleteAllByAttribute('rightID', $this->getOID());
  616.                     self::$logger->debug('...done deleting!');
  617.                 }
  618.                     
  619.                 // should set related field values to null (MySQL is doing this for us as-is)
  620.                 if($prop->getRelationType() == 'ONE-TO-MANY' && !$prop->getRelatedClass() == 'TagObject') {
  621.                     $relatedObjects = $prop->getRelatedObjects();
  622.                     
  623.                     foreach($relatedObjects as $object) {
  624.                         $object->set($prop->getRelatedClassField(), null);
  625.                         $object->save();
  626.                     }
  627.                 }
  628.                 
  629.                 // in the case of tags, we will always remove the related tags once the BO is deleted
  630.                 if($prop->getRelationType() == 'ONE-TO-MANY' && $prop->getRelatedClass() == 'TagObject') {
  631.                     // just making sure that the Relation is set to current OID as its transient
  632.                     $prop->setValue($this->getOID());
  633.                     $relatedObjects = $prop->getRelatedObjects();                    
  634.                     
  635.                     foreach($relatedObjects as $object) {                        
  636.                         $object->delete();
  637.                     }
  638.                 }
  639.             }
  640.         }
  641.         
  642.         $provider = AlphaDAOProviderFactory::getInstance($config->get('sysDBProviderName'), $this);
  643.         $provider->delete();
  644.         
  645.         if($config->get('sysCacheProviderName') != '')
  646.             $this->removeFromCache();
  647.         
  648.         if(method_exists($this, 'after_delete_callback'))
  649.             $this->after_delete_callback();
  650.             
  651.         $this->clear();
  652.         self::$logger->debug('<<delete');
  653.     }
