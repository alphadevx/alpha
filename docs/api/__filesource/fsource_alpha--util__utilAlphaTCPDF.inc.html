  67.      * @param float $y Ordinate of the upper-left corner.

  75.      * @param string $palign Allows to center or align the image on the current line. Possible values are:<ul><li>L : left align</li><li>C : center</li><li>R : right align</li><li>'' : empty string : left for LTR or right for RTL</li></ul>
  76.      * @param boolean $ismask true if this image is a mask, false otherwise

  78.      * @param mixed $border Indicates if borders must be drawn around the image. The value can be either a number:<ul><li>0: no border (default)</li><li>1: frame</li></ul>or a string containing some or all of the following characters (in any order):<ul><li>L: left</li><li>T: top</li><li>R: right</li><li>B: bottom</li></ul>

  110.     /**

  112.      * 
  113.      * @param array $dom html dom array
  114.      * @param int $key current element id
  115.      * @param boolean $cell if true add the default cMargin space to each new line (default false).

  118.     protected function openHTMLTagHandler(&$dom, $key, $cell=false) {
  119.         $tag = $dom[$key];
  120.         $parent = $dom[($dom[$key]['parent'])];
  121.         $firstorlast = ($key == 1);
  122.         // check for text direction attribute
  123.         if (isset($tag['attribute']['dir'])) {
  124.             $this->tmprtl = $tag['attribute']['dir'] == 'rtl' ? 'R' : 'L';
  125.         } else {
  126.             $this->tmprtl = false;
  127.         }
  128.         //Opening tag
  129.         switch($tag['value']) {
  130.             case 'table': {
  131.                 $cp = 0;
  132.                 $cs = 0;
  133.                 $dom[$key]['rowspans'] = array();
  134.                 if (!$this->empty_string($dom[$key]['thead'])) {
  135.                     // set table header
  136.                     $this->thead = $dom[$key]['thead'];
  137.                 }
  138.                 if (isset($tag['attribute']['cellpadding'])) {
  139.                     $cp = $this->getHTMLUnitToUnits($tag['attribute']['cellpadding'], 1, 'px');
  140.                     $this->oldcMargin = $this->cMargin;
  141.                     $this->cMargin = $cp;
  142.                 }
  143.                 if (isset($tag['attribute']['cellspacing'])) {
  144.                     $cs = $this->getHTMLUnitToUnits($tag['attribute']['cellspacing'], 1, 'px');
  145.                 }
  146.                 $this->checkPageBreak((2 * $cp) + (2 * $cs) + $this->lasth);
  147.                 break;
  148.             }
  149.             case 'tr': {
  150.                 // array of columns positions
  151.                 $dom[$key]['cellpos'] = array();
  152.                 break;
  153.             }
  154.             case 'hr': {
  155.                 $this->addHTMLVertSpace(1, $cell, '', $firstorlast, $tag['value'], false);
  156.                 $this->htmlvspace = 0;
  157.                 $wtmp = $this->w - $this->lMargin - $this->rMargin;
  158.                 if ((isset($tag['attribute']['width'])) AND ($tag['attribute']['width'] != '')) {
  159.                     $hrWidth = $this->getHTMLUnitToUnits($tag['attribute']['width'], $wtmp, 'px');
  160.                 } else {
  161.                     $hrWidth = $wtmp;
  162.                 }
  163.                 $x = $this->GetX();
  164.                 $y = $this->GetY();
  165.                 $prevlinewidth = $this->GetLineWidth();
  166.                 $this->Line($x, $y, $x + $hrWidth, $y);
  167.                 $this->SetLineWidth($prevlinewidth);
  168.                 $this->addHTMLVertSpace(1, $cell, '', !isset($dom[($key + 1)]), $tag['value'], false);
  169.                 break;
  170.             }
  171.             case 'a': {
  172.                 if (array_key_exists('href', $tag['attribute'])) {
  173.                     $this->HREF['url'] = $tag['attribute']['href'];
  174.                 }
  175.                 $this->HREF['color'] = $this->htmlLinkColorArray;
  176.                 $this->HREF['style'] = $this->htmlLinkFontStyle;
  177.                 if (array_key_exists('style', $tag['attribute'])) {
  178.                     // get style attributes
  179.                     preg_match_all('/([^;:\s]*):([^;]*)/', $tag['attribute']['style'], $style_array, PREG_PATTERN_ORDER);
  180.                     $astyle = array();
  181.                     while (list($id, $name) = each($style_array[1])) {
  182.                         $name = strtolower($name);
  183.                         $astyle[$name] = trim($style_array[2][$id]);
  184.                     }
  185.                     if (isset($astyle['color'])) {
  186.                         $this->HREF['color'] = $this->convertHTMLColorToDec($astyle['color']);
  187.                     }
  188.                     if (isset($astyle['text-decoration'])) {
  189.                         $this->HREF['style'] = '';
  190.                         $decors = explode(' ', strtolower($astyle['text-decoration']));
  191.                         foreach ($decors as $dec) {
  192.                             $dec = trim($dec);
  193.                             if (!$this->empty_string($dec)) {
  194.                                 if ($dec{0} == 'u') {
  195.                                     $this->HREF['style'] .= 'U';
  196.                                 } elseif ($dec{0} == 'l') {
  197.                                     $this->HREF['style'] .= 'D';
  198.                                 }
  199.                             }
  200.                         }
  201.                     }
  202.                 }        
  203.                 break;
  204.             }
  205.             case 'img': {
  206.                 if (isset($tag['attribute']['src'])) {
  207.                     // replace relative path with real server path
  208.                     if ($tag['attribute']['src'][0] == '/') {
  209.                         $tag['attribute']['src'] = $_SERVER['DOCUMENT_ROOT'].$tag['attribute']['src'];
  210.                     }                        
  211.                     $tag['attribute']['src'] = urldecode($tag['attribute']['src']);
  212.                     //$tag['attribute']['src'] = str_replace(K_PATH_URL, K_PATH_MAIN, $tag['attribute']['src']);
  213.                     if (!isset($tag['attribute']['width'])) {
  214.                         $tag['attribute']['width'] = 0;
  215.                     }
  216.                     if (!isset($tag['attribute']['height'])) {
  217.                         $tag['attribute']['height'] = 0;
  218.                     }
  219.                     //if (!isset($tag['attribute']['align'])) {
  220.                         // the only alignment supported is "bottom"
  221.                         // further development is required for other modes.
  222.                         $tag['attribute']['align'] = 'bottom';
  223.                     //} 
  224.                     switch($tag['attribute']['align']) {
  225.                         case 'top': {
  226.                         $align = 'T';
  227.                             break;
  228.                         }
  229.                         case 'middle': {
  230.                             $align = 'M';
  231.                             break;
  232.                         }
  233.                         case 'bottom': {
  234.                             $align = 'B';
  235.                             break;
  236.                         }
  237.                         default: {
  238.                             $align = 'B';
  239.                             break;
  240.                         }
  241.                     }
  242.                     $fileinfo = pathinfo($tag['attribute']['src']);
  243.                     if (isset($fileinfo['extension']) AND (!$this->empty_string($fileinfo['extension']))) {
  244.                         $type = strtolower($fileinfo['extension']);
  245.                     }else{
  246.                         $type = '';
  247.                     }
  248.                     $prevy = $this->y;
  249.                     $xpos = $this->GetX();
  250.                     if (isset($dom[($key - 1)]) AND ($dom[($key - 1)]['value'] == ' ')) {
  251.                         if ($this->rtl) {
  252.                             $xpos += $this->GetStringWidth(' ');
  253.                         } else {
  254.                             $xpos -= $this->GetStringWidth(' ');
  255.                         }
  256.                     }
  257.                     $imglink = '';
  258.                     if (isset($this->HREF['url']) AND !$this->empty_string($this->HREF['url'])) {
  259.                         $imglink = $this->HREF['url'];
  260.                         if ($imglink{0} == '#') {
  261.                             // convert url to internal link
  262.                             $page = intval(substr($imglink, 1));
  263.                             $imglink = $this->AddLink();
  264.                             $this->SetLink($imglink, 0, $page);
  265.                         }
  266.                     }
  267.                     $border = 0;
  268.                     if (isset($tag['attribute']['border']) AND !empty($tag['attribute']['border'])) {
  269.                         // currently only support 1 (frame) or a combination of 'LTRB'
  270.                         $border = $tag['attribute']['border'];
  271.                     }
  272.                     if (isset($tag['attribute']['width'])) {
  273.                         $iw = $this->getHTMLUnitToUnits($tag['attribute']['width'], 1, 'px', false);
  274.                     }
  275.                     if (isset($tag['attribute']['height'])) {
  276.                         $ih = $this->getHTMLUnitToUnits($tag['attribute']['height'], 1, 'px', false);
  277.                     }
  278.                     if (($type == 'eps') OR ($type == 'ai')) {
  279.                         $this->ImageEps($tag['attribute']['src'], $xpos, $this->GetY(), $iw, $ih, $imglink, true, $align, '', $border);
  280.                     } else {
  281.                         $this->Image($tag['attribute']['src'], $xpos, $this->GetY(), $iw, $ih, '', $imglink, $align, false, 300, '', false, false, $border);
  282.                     }
  283.                     switch($align) {
  284.                         case 'T': {
  285.                             $this->y = $prevy;
  286.                             break;
  287.                         }
  288.                         case 'M': {
  289.                             $this->y = (($this->img_rb_y + $prevy - ($tag['fontsize'] / $this->k)) / 2) ;
  290.                             break;
  291.                         }
  292.                         case 'B': {
  293.                             $this->y = $this->img_rb_y - ($tag['fontsize'] / $this->k);
  294.                             break;
  295.                         }
  296.                     }
  297.                 }
  298.                 break;
  299.             }
  300.             case 'dl': {
  301.                 ++$this->listnum;
  302.                 $this->addHTMLVertSpace(0, $cell, '', $firstorlast, $tag['value'], false);
  303.                 break;
  304.             }
  305.             case 'dt': {
  306.                 $this->addHTMLVertSpace(1, $cell, '', $firstorlast, $tag['value'], false);
  307.                 break;
  308.             }
  309.             case 'dd': {
  310.                 if ($this->rtl) {
  311.                     $this->rMargin += $this->listindent;
  312.                 } else {
  313.                     $this->lMargin += $this->listindent;
  314.                 }
  315.                 $this->addHTMLVertSpace(1, $cell, '', $firstorlast, $tag['value'], false);
  316.                 break;
  317.             }
  318.             case 'ul':
  319.             case 'ol': {
  320.                 $this->addHTMLVertSpace(0, $cell, '', $firstorlast, $tag['value'], false);
  321.                 $this->htmlvspace = 0;
  322.                 ++$this->listnum;
  323.                 if ($tag['value'] == 'ol') {
  324.                     $this->listordered[$this->listnum] = true;
  325.                 } else {
  326.                     $this->listordered[$this->listnum] = false;
  327.                 }
  328.                 if (isset($tag['attribute']['start'])) {
  329.                     $this->listcount[$this->listnum] = intval($tag['attribute']['start']) - 1;
  330.                 } else {
  331.                     $this->listcount[$this->listnum] = 0;
