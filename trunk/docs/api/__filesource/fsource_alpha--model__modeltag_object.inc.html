  214.      * and returns an array of tag_object instances.

  223.     public static function tokenize($content, $taggedClass='', $taggedOID='', $applyStopwords=true) {
  224.         if(self::$logger == null)
  225.             self::$logger = new Logger('tag_object');
  226.             
  227.         global $config;
  228.         
  229.         // apply stop words
  230.         $lowerWords = preg_split("/[\s,.:]+/", $content);
  231.         array_walk($lowerWords, 'tag_object::lowercaseArrayElement');
  232.         if($applyStopwords) {        
  233.             $stopwords = file($config->get('sysRoot').'config/stopwords.ini', FILE_IGNORE_NEW_LINES);
  234.             array_walk($stopwords, 'tag_object::lowercaseArrayElement');
  235.             $filtered = array_diff($lowerWords, $stopwords);
  236.         }else{
  237.             $filtered = $lowerWords;
  238.         }
  239.         
  240.         $tagObjects = array();
  241.         $tagContents = array();
  242.         foreach($filtered as $tagContent) {
  243.             // we only want to create word tags
  244.             if(Validator::isAlpha($tagContent)) {
  245.                 // just making sure that we haven't added this one in already
  246.                 if(!in_array($tagContent, $tagContents) && !empty($tagContent)) {
  247.                     $tag = new tag_object();
  248.                     $tag->set('content', trim(strtolower($tagContent)));
  249.                     if(!empty($taggedClass))
  250.                         $tag->set('taggedClass', $taggedClass);
  251.                     if(!empty($taggedOID))
  252.                         $tag->set('taggedOID', $taggedOID);
  253.                     
  254.                     array_push($tagObjects, $tag);
  255.                     array_push($tagContents, $tagContent);
  256.                 }
  257.             }
  258.         }
  259.         
  260.         self::$logger->debug('Tags generated: ['.print_r($tagContents, true).']');
  261.         return $tagObjects;
  262.     }
