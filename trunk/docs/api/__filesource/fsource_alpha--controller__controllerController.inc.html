  597.         $dirtyObjects = $this->getDirtyObjects();

  601.                 $dirtyObjects[$i]->save();

  606.                 return;
  607.             }catch (LockingException $e) {

  619.         $_SESSION['unitOfWork'] = null;

  812.     /**

  814.      * fields which aim to ensure that a post to the controller is being sent from
  815.      * the same server that is hosting it.

  819.     public static function checkSecurityFields() {
  820.         if(self::$logger == null)
  821.             self::$logger = new Logger('Controller');
  822.         self::$logger->debug('>>checkSecurityFields()');
  823.         
  824.         // the server hostname + today's date
  825.         $var1 = md5($_SERVER['HTTP_HOST'].date("Ymd"));
  826.         // the server's IP plus $var1
  827.         $var2 = md5($_SERVER['REMOTE_ADDR'].$var1);
  828.         
  829.         if(empty($_REQUEST['var1']) || empty($_REQUEST['var2'])) {
  830.             self::$logger->debug('<<checkSecurityFields [false]');
  831.             return false;
  832.         }
  833.         
  834.         if ($var1 == $_REQUEST['var1'] && $var2 == $_REQUEST['var2']) {            
  835.             self::$logger->debug('<<checkSecurityFields [true]');
  836.             return true;
  837.         }else{
  838.             /*
  839.              * Here we are implementing a "grace period" of one hour if the time is < 1:00AM, we will accept
  840.              * a match for yesterday's date in the security fields
  841.              * 
  842.              */            
  843.             
  844.             // the server hostname + today's date less 1 hour (i.e. yesterday where time is < 1:00AM)
  845.             $var1 = md5($_SERVER['HTTP_HOST'].date("Ymd", (time()-3600)));
  846.             // the server's IP plus $var1
  847.             $var2 = md5($_SERVER['REMOTE_ADDR'].$var1);
  848.             
  849.             if ($var1 == $_REQUEST['var1'] && $var2 == $_REQUEST['var2']) {                
  850.                 self::$logger->debug('<<checkSecurityFields [true]');
  851.                 return true;
  852.             }else{
  853.                 self::$logger->debug('<<checkSecurityFields [false]');
  854.                 return false;
  855.             }
  856.         }
