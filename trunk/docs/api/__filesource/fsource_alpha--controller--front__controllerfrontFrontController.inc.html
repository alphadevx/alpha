  218.         $td = mcrypt_module_open ('tripledes', '', 'ecb', '');

  368.             if($config->get('sysEnableClientTempBlacklistFiler')) {

  498.     /**
  499.      * Handles all of the rules for mod_rewrite style URL parsing
  500.      * 
  501.      * @since 1.0
  502.      */
  503.     private function handleModRewriteRequests() {
  504.         self::$logger->debug('>>handleModRewriteRequests');
  505.         global $config;
  506.         
  507.         // strip off the system URL from the request URL
  508.         $request = str_replace($config->get('sysURL'), '', 'http://'.$_SERVER['HTTP_HOST'].$_SERVER['REQUEST_URI']);
  509.         self::$logger->debug('$request is ['.$request.']');
  510.         $params = explode('/', $request);
  511.         self::$logger->debug('$params are ['.print_r($params, true).']');
  512.         
  513.         try {
  514.             // first param will always be the controller alias
  515.             if(empty($this->currentAlias) && !empty($params[0]))
  516.                 $this->currentAlias = $params[0];
  517.  
  518.             // check to see if we can load the page controller without an alias
  519.             AlphaController::loadControllerDef($params[0]);
  520.             self::$logger->debug('Page controller name set on the request URL is ['.$params[0].']');
  521.             $this->pageController = $params[0];
  522.         }catch (IllegalArguementException $iae) {
  523.             // handle request with alias        
  524.             self::$logger->debug('The supplied controller alias is ['.$this->currentAlias.']');
  525.             
  526.             // check to see if the controller is an alias for something
  527.             if($this->checkAlias($this->currentAlias)) {
  528.                 $this->pageController = $this->getAliasController($this->currentAlias);
  529.                 self::$logger->debug('Page controller name obtained from the URL alias is ['.$this->pageController.']');
  530.                 
  531.                 if(isset($params[1])) {
  532.                     if(!empty($_POST))
  533.                         $_REQUEST[$this->getAliasParam($this->currentAlias)] = $params[1];
  534.                     else
  535.                         $_GET[$this->getAliasParam($this->currentAlias)] = $params[1];
  536.                 }
  537.             }
  538.         }
  539.         
  540.         self::$logger->debug('$params are ['.print_r($params, true).']');
  541.         self::$logger->debug('currentAlias is ['.$this->currentAlias.']');
  542.         
  543.         // now populate the _GET vars
  544.         if($this->currentAlias == 'tk') {
  545.             self::$logger->debug('Setting the GET vars for a mod_rewrite request with a tk param');
  546.             $this->setEncrypt(true);
  547.             $this->queryString = FrontController::decodeQueryParams($params[1]);
  548.             $this->populateGetVars();
  549.             $this->pageController = $_GET['act'];
  550.         }else{
  551.             for($i = 1; $i < count($params); $i+=2) {
  552.                 if(isset($params[$i+1])) {
  553.                     if(!empty($_POST))
  554.                         $_REQUEST[$params[$i]] = $params[$i+1];
  555.                     else            
  556.                         $_GET[$params[$i]] = $params[$i+1];
  557.                 }
  558.             }
  559.         }
  560.         
  561.         self::$logger->debug('$_GET is ['.print_r($_GET, true).']');        
  562.         self::$logger->debug('<<handleModRewriteRequests');
  563.     }
  564.
