  259.     function html2rgb($colour)
  260.     {
  261.         if ($colour[0] == '#')
  262.             $colour = substr($colour, 1);
  263.     
  264.         if (strlen($colour) == 6)
  265.             list($r, $g, $b) = array($colour[0].$colour[1],
  266.                                      $colour[2].$colour[3],
  267.                                      $colour[4].$colour[5]);
  268.         elseif (strlen($colour) == 3)

  273.         $r = hexdec($r); $g = hexdec($g); $b = hexdec($b);

  314.             
  315.             if ($isMask){

  344.     // needs GD 2.x extension
  345.     // pixel-wise operation, not very fast
  346.     function ImagePngWithAlpha($file,$x,$y,$w=0,$h=0,$link='')
  347.     {
  348.         global $config;
  349.         
  350.         $tmp_alpha = tempnam($config->get('sysRoot').'cache/images/', 'mska');
  351.         $this->tmpFiles[] = $tmp_alpha;
  352.         $tmp_plain = tempnam($config->get('sysRoot').'cache/images/', 'mskp');
  353.         $this->tmpFiles[] = $tmp_plain;
  354.         
  355.         list($wpx, $hpx) = getimagesize($file);
  356.         $img = imagecreatefrompng($file);
  357.         $alpha_img = imagecreate( $wpx, $hpx );
  358.         
  359.         // generate gray scale pallete
  360.         for($c=0;$c<256;$c++) ImageColorAllocate($alpha_img, $c, $c, $c);
  361.         
  362.         // extract alpha channel
  363.         $xpx=0;
  364.         while ($xpx<$wpx){
  365.             $ypx = 0;
  366.             while ($ypx<$hpx){
  367.                 $color_index = imagecolorat($img, $xpx, $ypx);
  368.                 $col = imagecolorsforindex($img, $color_index);
  369.                 imagesetpixel($alpha_img, $xpx, $ypx, $this->_gamma( (127-$col['alpha'])*255/127)  );
  370.             ++$ypx;
  371.             }
  372.             ++$xpx;
  373.         }
  374.     
  375.         imagepng($alpha_img, $tmp_alpha);
  376.         imagedestroy($alpha_img);
  377.         
  378.         // extract image without alpha channel
  379.         $plain_img = imagecreatetruecolor ( $wpx, $hpx );
  380.         imagecopy ($plain_img, $img, 0, 0, 0, 0, $wpx, $hpx );
  381.         imagepng($plain_img, $tmp_plain);
  382.         imagedestroy($plain_img);
  383.         
  384.         //first embed mask image (w, h, x, will be ignored)
  385.         $maskImg = $this->Image($tmp_alpha, 0,0,0,0, 'PNG', '', true);
  386.         
  387.         //embed image, masked with previously embedded mask
  388.         $this->Image($tmp_plain,$x,$y,$w,$h,'PNG',$link, false, $maskImg);
  389.     }
