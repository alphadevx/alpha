  26. function drawImage($source, $width, $height, $sourceType, $quality) {
  27.     /* returns a PNG images to the web browser
  28.     Parameters:
  29.     $source: the file location of the source images
  30.     $width: the width of the outputted image
  31.     $height: the height of the images
  32.     $sourceType: the type of image that that source is (options are "jpeg","gif", and "png")
  33.

  38.     // now get the old image
  39.     switch ($sourceType) {
  40.         case "gif":
  41.             $old_image = imagecreatefromgif($config->get('sysRoot').$source);
  42.         break;
  43.         case "jpg":
  44.             $old_image = imagecreatefromjpeg($config->get('sysRoot').$source);
  45.         break;
  46.         case "png":
  47.             $old_image = imagecreatefrompng($config->get('sysRoot').$source);
  48.         break;
  49.     }
  50.     
  51.     if (!$old_image) { /* See if it failed */ 
  52.         $im  = imagecreatetruecolor($width,$height); /* Create a blank image */ 
  53.         $bgc = imagecolorallocate($im, 255, 255, 255); 
  54.         $tc  = imagecolorallocate($im, 0, 0, 0); 
  55.         imagefilledrectangle($im, 0, 0, $width, $height, $bgc); 
  56.         
  57.         imagestring($im, 1, 5, 5, "Error loading $source", $tc);

  60.         imagedestroy($im);

  64.         $oldHeight = imagesy($old_image);
  65.     
  66.         // now create the new image
  67.         $new_image = imagecreatetruecolor($width,$height);

  69.         // copy the old image to the new image (in memory, not the file!)

  71.             
  72.         header("Content-Type: image/jpeg");

  74.         imagedestroy($old_image);
  75.         imagedestroy($new_image);
